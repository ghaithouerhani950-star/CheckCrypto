async function updateMarket() { if (document.hidden) return; try { const query = coins.map(c => `"${c.sym}"`).join(','); const response = await fetch(`https://api.binance.com/api/v3/ticker/price?symbols=[${query}]`); const data = await response.json(); data.forEach(item => { const id = item.symbol.replace('USDT', ''); const p = parseFloat(item.price); const oldP = prices[item.symbol]; if (oldP !== 0 && oldP !== p) { const isUp = p > oldP; const st = document.getElementById(`s-${id}`); const card = document.getElementById(`card-${id}`); st.innerText = isUp ? "▲ UP" : "▼ DOWN"; st.className = "price-chg " + (isUp ? "up" : "down"); card.classList.add(isUp ? 'flash-up' : 'flash-down'); setTimeout(() => card.classList.remove('flash-up', 'flash-down'), 400); } document.getElementById(`p-${id}`).innerText = p < 1 ? p.toFixed(4) : p.toLocaleString(undefined, {minimumFractionDigits: 2}); prices[item.symbol] = p; }); document.getElementById('status').innerText = "MARKET LIVE • " + new Date().toLocaleTimeString(); calc(); } catch (e) { document.getElementById('status').innerText = "RECONNECTING..."; } } function calc() { const v = parseFloat(document.getElementById('amt').value); const coin = sel.value; const p = prices[coin + 'USDT']; if (v > 0 && p > 0) document.getElementById('res').innerText = `${(v / p).toFixed(6)} ${coin}`; } function openModal() { document.getElementById('aboutModal').style.display = 'flex'; } function closeModal() { document.getElementById('aboutModal').style.display = 'none'; } window.onclick = (e) => { if (e.target == document.getElementById('aboutModal')) closeModal(); } document.getElementById('res').addEventListener('click', function() { const val = this.innerText.split(' ')[0]; if (val !== "0.00") { navigator.clipboard.writeText(val); const t = document.getElementById('toast'); t.style.display = 'block'; setTimeout(() => t.style.display = 'none', 1500); } }); document.getElementById('amt').addEventListener('input', calc); sel.addEventListener('change', calc); init(); updateMarket(); setInterval(updateMarket, 4000);
